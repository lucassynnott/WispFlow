### Time-Based Greeting Logic
```swift
private var greetingMessage: String {
    let hour = Calendar.current.component(.hour, from: Date())
    if hour < 12 {
        return "Good morning."
    } else if hour < 17 {
        return "Good afternoon."
    } else {
        return "Good evening."
    }
}
```
